% -- Listes et énumération --
%
% Il peut être intéressant de lister ou de numéroter les objets sous latex.
% Comme pour les section, les listes ou énumérations peuvent s'indenter. Une
% liste ou une énumération est un environnement, elles sont donc contenues entre
% des commandes "begin" et "end"
%
% Chaque élément la liste ou de l'énumération doit débuter par la commande \item.
% Vous pouvez mettre pratiquement n'importe quoi à la suite, tant que vous 
% n'écrivez pas un autre \item vous restez dans le même item. Du fait que la 
% liste et l'énumération utilisent tous deux les "items", il est facile de 
% convertir une énumération en liste et vice-versa.
%
% Pour une liste :
\begin{itemize}
\item Premier Item
\item Deuxième Item
	\begin{itemize}
	\item Sous-item 1
	\item Sous-item 2
	\end{itemize}
\item Troisième Item
\end{itemize}

\bigskip\bigskip

% Et pour une énumération : (Remarquez que le style d'énumération change)
\begin{enumerate}
\item Premier Item
\item Deuxième Item
	\begin{enumerate}
	\item Sous-item 1
	\item Sous-item 2
	\end{enumerate}
\item Troisième Item
\end{enumerate}

\bigskip\bigskip

% On peut aussi mélanger les deux.
\begin{itemize}
\item Premier Item
\item Deuxième Item
	\begin{enumerate}
	\item Sous-item énuméré 1
	\item Sous-item énuméré 2
	\end{enumerate}
\item Troisième Item
\end{itemize}

% -- Espacements du texte --
%
% LateX contrôle généralement bien les espacement et les tailles de texte. Malgré
% tout, il peut parfois être intéressant de pouvoir contrôler les espacements et 
% autres caractères vides sous latex. Pour ce faire, on peut utiliser les 
% commandes suivantes :
%
%	\bigskip ou \smallskip :	Produit un espacement vertical de taille plus
%										moins grosse constante.
%
%	\\ :	Placé à la fin d'une ligne, les barres obliques inversées doubles
%			forcent latex à crééer une nouvelle ligne vide.
%
%	\vspace{espacement} :	Produit un espacement de la longueur demandée avec
%									l'unité demandée (pt, in, mm, cm, etc.). Il existe
%									ausi un équivalent horizontal \hspace
%
%	\noindent :	Enlève l'indentation de première ligne d'un paragraphe lorsque
%					placé avant celui-ci.
%
%	\vfill :	Place autant d'espace blanc vertical que possible entre deux
%				éléments. ( \hfill est aussi disponible pour faire la même chose
%				horizontalement)
%
% Finalement, pour séparer deux paragraphe, il faut simplement laisser deux 
% lignes blanches. Pour que les paragraphes aient un espace vide, utiliser les
% \\.

% -- Alignement du texte --
% 
% Sous LateX, l'alignement du texte s'effectue à l'aide des environnements
% "flushleft", "flushright", "center" et "quote".
%
%	flushleft et flushright :	Alignent le texte respectivement à gauche et à
%										droite.
%
%	center :	Aligne le texte au centre
%
%	quote :	Aligne le texte à gauche avec un certain espace avant et après
%				le texte.
\newpage

\begin{center}
Ce texte est aligné au centre.
\end{center}

\begin{flushleft}
Ce texte est aligné à gauche. L'alignement à gauche est déjà activé par défaut.
\end{flushleft}

\begin{flushright}
Ce texte est aligné à droite.
\end{flushright}



\begin{quote}
Ce texte est une citation, il est donc aligné à gauche mais avec un espacement 
avant le texte. Je suis en train de tapper une longue phrase qui veut pas dire
grand chose uniquement pour que vous puissiez remarquer l'effet sur plusieurs
lignes et constater de celui-ci par vous-même. Booya.
\end{quote}

% -- Tableaux --
%
% Une dernière méthode pour organiser votre document est de présenter
% l'information sous la forme de tableaux. La création d'un tableau s'effectue
% dans l'environnement "tabular" et suit la syntaxe suivante :
%
% \begin{tabular}{organisation des colonnes}
%		tableau ici
% \end{tabular}
%
% La section pour l'organisation des colonnes permet d'indiquer comment chaque
% colonnes seront alignées dans le document, et comment les colonnes seront
% séparées. Il existe 3 types d'alignement pour les colonnes :
%
%	- r (right)
%	- l (left)
%	- c (center)
%
% Pour indiquer comment le nombre de colonnes dans un tableau et leur alignement
% on écrit simplement les lettres une à la suite des autres entre le deuxième
% couple d'accolades après "tabular". Si on souhaite que les colonnes soient
% séparées par un trait, on écrit "|" entre deux lettres. Par exemple :
%
% \begin{tabular}{|c|c|l|} 
%
% Génèrerait un tableau à trois colonnes, toutes séparées par des lignes
% verticales, les deux premières étant centrées et la troisième étant alignée
% à gauche.
%
% Finalement, lorsqu'on rempli le tableau, chaque ligne du tableau doit terminer
% par un "\\". Chaque cellule d'une ligne est quant à elle séparée d'un "&" et
% il est possible d'insérer un trait horizontal entre chaque ligne en utilisant
% la commande "\hline".
%
% Des exemples concret :
%
\newpage

\begin{tabular}{|r|c l|}
	\hline
	1.1 & 1.2 & 1.3\\ 
	\hline
	2.1 & 2.2 &2.3\\
	\hline
	3.1 & 3.2 & 3.3\\
	\hline
\end{tabular}

\vspace{30mm}

\end{tabular}
